using UnityEngine;

public class LightController : MonoBehaviour
{
    Light l;
    Color c;
    float i;

    void Start()
    {
        l = GetComponent<Light>();
        c = l.color;
        i = l.intensity;
    }

    void Update()
    {
        if (Input.GetKeyDown(KeyCode.Alpha1)) { c = Color.yellow; i = 2f; }
        if (Input.GetKeyDown(KeyCode.Alpha2)) { c = Color.cyan;   i = 4f; }
        if (Input.GetKeyDown(KeyCode.Alpha3)) { c = Color.green;  i = 1f; }

        l.color = Color.Lerp(l.color, c, Time.deltaTime * 2);
        l.intensity = Mathf.Lerp(l.intensity, i, Time.deltaTime * 2);
    }
}

using UnityEngine;

public class Controller : MonoBehaviour
{
    public Animator animator;
    public ParticleSystem fire;

    void Update()
    {
        // 🔁 ROTATION / ANIMATION (W key)
        if (Input.GetKey(KeyCode.W))
        {
            animator.SetFloat("Speed", 1);
        }
        else
        {
            animator.SetFloat("Speed", 0);
        }

        // ✨ PARTICLE TOGGLE (Space key)
        if (Input.GetKeyDown(KeyCode.Space))
        {
            if (fire.isPlaying)
                fire.Stop();
            else
                fire.Play();
        }
    }
}

LAB 2: Basic Player Movement (Updated Steps)
🧠 Objective

Create a player (Capsule) that can:

Move forward/back (W/S)
Move left/right (A/D)
Detect collision with an obstacle
🏗️ STEP 1: Create Scene
Open Unity → New 3D Scene
In Hierarchy:
Right click → 3D Object → Plane     (Ground)
Right click → 3D Object → Capsule   (rename: Player)
Right click → 3D Object → Cube      (Obstacle)
🏷️ STEP 2: Add Tag to Obstacle
Select Cube
In Inspector → Tag → Add Tag
Click +
Create:
Obstacle
Go back → Select Cube again → assign:
Tag = Obstacle
⚙️ STEP 3: Add Rigidbody to Player

Select Capsule (Player)

Add Component → Rigidbody
Set:
Use Gravity ✅
Freeze Rotation X, Y, Z ✅

👉 Prevents player from falling/tilting

💻 STEP 4: Create Script

Right click in Project → Create → C# Script → Lab2.cs
using UnityEngine;

public class Lab2 : MonoBehaviour
{
    public float speed = 5f;

    void Update()
    {
        float moveX = Input.GetAxis("Horizontal"); // A/D
        float moveZ = Input.GetAxis("Vertical");   // W/S

        Vector3 move = new Vector3(moveX, 0f, moveZ);

        transform.Translate(move * speed * Time.deltaTime);
    }

    void OnCollisionEnter(Collision collision)
    {
        if (collision.gameObject.CompareTag("Obstacle"))
        {
            Debug.Log("Obstacle Found!");
        }
    }
}
